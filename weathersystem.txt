	clear_points()
	add_point(start_pos)
	
	var current_pos = start_pos
	var direction = (end_pos - start_pos).normalized()
	var total_distance = start_pos.distance_to(end_pos)
	var segment_length = total_distance / segments
	
	for i in range(1, segments):
		# Calculate progress along the main line path
		var progress = float(i) / float(segments)
		var target_point = start_pos + direction * (segment_length * i)
		
		# Add perpendicular jagged noise displacement
		var normal = Vector2(-direction.y, direction.x)
		var sway = randf_range(-displacement, displacement) * (1.0 - progress * 0.5)
		var next_point = target_point + normal * sway
		
		add_point(next_point)
		
		# 15% chance to create a smaller sub-branch splitting off
		if randf() < 0.15 and i < segments - 3:
			_create_branch(next_point, direction, segments - i, displacement * 0.6)
			
	add_point(end_pos)
	
	# Animate the lightning bolt fading out quickly
	var tween = create_tween()
	tween.tween_property(self, "modulate:a", 0.0, randf_range(0.15, 0.3)).set_trans(Tween.TRANS_EXPO)
	tween.finished.connect(queue_free)

	# Fetch values directly out of our centralized cross-scene Rendering Server Globals
	var active_puddles: float = RenderingServer.global_shader_parameter_get("puddle_depth")
	var active_snow: float = RenderingServer.global_shader_parameter_get("snow_accumulation")
	
	# 1. Apply Downward Gravity Forces
	if not is_on_floor():
		velocity.y += gravity * delta

	# 2. Process Input Jumps
	if Input.is_action_just_pressed("ui_accept") and is_on_floor():
		velocity.y = JUMP_VELOCITY

	# 3. Read Direct Horizontal Input Vector Axes
	var direction := Input.get_axis("ui_left", "ui_right")
	
	# 4. Calculate Friction and Acceleration dynamically based on ground environment conditions
	var target_accel = NORMAL_ACCEL
	var target_friction = NORMAL_FRICTION
	
	if is_on_floor():
		if active_snow > 0.1:
			# Snow gives low acceleration traction and massive slippage drifts
